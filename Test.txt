function deneme(model,ignoreProperties){
  for(var prop in model) {
    if(ignoreProperties != null || ignoreProperties != undefined){
      if(ignoreProperties.indexOf(prop) == -1){
        model[prop] = encodeURIComponent(model[prop]);
      }
    }
    else{
      model[prop] = encodeURIComponent(model[prop]);
    }
  }

  console.log(model);
}
